# docker-compose.yml
version: '3.8'

services:
  backend:
    build: ./backend
    container_name: edulearn-backend
    restart: unless-stopped
    environment:
      - NODE_ENV=production
      - MONGODB_URI=mongodb://mongo:27017/edulearn
      - JWT_ACCESS_SECRET=${JWT_ACCESS_SECRET}
      - JWT_REFRESH_SECRET=${JWT_REFRESH_SECRET}
      - REDIS_URL=redis://redis:6379
    volumes:
      - uploads:/app/uploads
      - ./backend/src:/app/src
      - /app/src/node_modules
    depends_on:
      - mongo
      - redis
    networks:
      - edulearn-network

  frontend:
    build: ./frontend
    container_name: edulearn-frontend
    restart: unless-stopped
    environment:
      - REACT_APP_API_URL=https://edu.learn.com/api
    volumes:
      - ./frontend/src:/app/src
      - /app/src/node_modules
    networks:
      - edulearn-network

  nginx:
    image: nginx:alpine
    container_name: edulearn-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./nginx/ssl:/etc/nginx/ssl
      - uploads:/app/uploads
    depends_on:
      - backend
      - frontend
    networks:
      - edulearn-network

  mongo:
    image: mongo:5
    container_name: edulearn-mongo
    restart: unless-stopped
    volumes:
      - mongo-data:/data/db
    environment:
      - MONGO_INITDB_DATABASE=edulearn
    networks:
      - edulearn-network

  redis:
    image: redis:7-alpine
    container_name: edulearn-redis
    restart: unless-stopped
    networks:
      - edulearn-network

networks:
  edulearn-network:
    driver: bridge

volumes:
  uploads:
  mongo-data:
